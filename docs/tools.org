* 概述
** 工具系统目标
   1. **使用统计**：记录每个工具的使用频率和成功率
   2. **性能分析**：了解哪些工具最常用，哪些很少使用
   3. **项目分析**：分析不同项目的工具使用模式
   4. **优化决策**：基于使用数据决定工具的增强或废弃

** 核心功能
   - 工具使用记录和统计
   - 工具使用情况可视化
   - 项目工具使用分析
   - 工具性能监控

* 数据库设计
** 工具表 (tools)
   #+BEGIN_SRC sql
     CREATE TABLE tools (
         id INTEGER PRIMARY KEY AUTOINCREMENT,
         name TEXT NOT NULL UNIQUE,           -- 工具名称
         description TEXT NOT NULL,           -- 工具描述
         category TEXT,                       -- 工具分类
         usage_count INTEGER DEFAULT 0,       -- 使用次数
         created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
         updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
     );
   #+END_SRC

** 工具使用记录表 (tool_usage)
   #+BEGIN_SRC sql
     CREATE TABLE tool_usage (
         id INTEGER PRIMARY KEY AUTOINCREMENT,
         project_hash TEXT NOT NULL,          -- 项目哈希
         tool_id INTEGER NOT NULL,            -- 工具ID
         used_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
         success BOOLEAN DEFAULT 1,           -- 是否成功
         error_msg TEXT,                      -- 错误信息（如果失败）
         FOREIGN KEY (tool_id) REFERENCES tools(id) ON DELETE CASCADE
     );
   #+END_SRC

** 索引设计
   #+BEGIN_SRC sql
     -- 工具分类索引
     CREATE INDEX idx_tools_category ON tools(category);
     
     -- 工具使用次数索引（用于排序）
     CREATE INDEX idx_tools_usage ON tools(usage_count DESC);
     
     -- 项目工具使用索引
     CREATE INDEX idx_tool_usage_project ON tool_usage(project_hash);
     
     -- 工具使用索引
     CREATE INDEX idx_tool_usage_tool ON tool_usage(tool_id);
     
     -- 时间索引（用于时间范围查询）
     CREATE INDEX idx_tool_usage_time ON tool_usage(used_at DESC);
   #+END_SRC

* 统计指标
** 基础指标
   1. **使用次数**：每个工具被调用的总次数
   2. **成功率**：工具执行成功的比例
   3. **使用频率**：单位时间内的使用次数
   4. **错误类型**：失败的原因分布

** 项目指标
   1. **项目工具使用分布**：不同项目的工具使用模式
   2. **项目活跃度**：基于工具使用频率的项目活跃度
   3. **项目类型识别**：基于工具使用模式识别项目类型

** 时间指标
   1. **趋势分析**：工具使用随时间的变化趋势
   2. **峰值识别**：识别工具使用的高峰时段
   3. **季节性分析**：工具使用的周期性模式

* 实现细节
** 工具调用流程
   #+BEGIN_SRC go
     func HandleToolCall(toolName string, projectRoot string, args json.RawMessage) (string, error) {
         // 1. 获取或创建工具记录
         toolID, err := db.GetOrCreateTool(toolName, description, category)
         
         // 2. 执行工具
         startTime := time.Now()
         result, err := toolHandlers[toolName](projectRoot, args)
         duration := time.Since(startTime)
         
         // 3. 记录使用情况
         success := err == nil
         errorMsg := ""
         if err != nil {
             errorMsg = err.Error()
         }
         
         projectHash := db.GetProjectHash(projectRoot)
         db.RecordToolUsage(toolID, projectHash, success, errorMsg)
         
         // 4. 返回结果
         return result, err
     }
   #+END_SRC

** 工具注册机制
   #+BEGIN_SRC go
     // 工具定义结构
     type ToolDef struct {
         Name        string
         Description string
         Category    string
         Handler     func(projectRoot string, args json.RawMessage) (string, error)
     }
     
     // 工具注册表
     var toolRegistry = map[string]ToolDef{}
     
     // 注册工具
     func RegisterTool(tool ToolDef) {
         toolRegistry[tool.Name] = tool
         
         // 自动注册到数据库
         db.GetOrCreateTool(tool.Name, tool.Description, tool.Category)
     }
     
     // 获取所有工具定义（用于API调用）
     func GetAllTools() []api.Tool {
         var tools []api.Tool
         for _, tool := range toolRegistry {
             tools = append(tools, tool.ToApiTool())
         }
         return tools
     }
   #+END_SRC
